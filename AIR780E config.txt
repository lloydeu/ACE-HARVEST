# Configuration
$portName = "COM5" 
$port = New-Object System.IO.Ports.SerialPort $portName, 115200, None, 8, One
$port.Open()

# 1. Set SMS to save to SIM memory instead of just displaying
# CNMI=2,1 means: Notify me of the index, and SAVE it to memory.
Write-Host "Configuring module to SAVE messages..." -ForegroundColor Cyan
$port.WriteLine("AT+CNMI=2,1,0,0,0`r")
Start-Sleep -m 500

# 2. Set default storage to SIM card
$port.WriteLine("AT+CPMS=`"SM`",`"SM`",`"SM`"`r")
Start-Sleep -m 500

$response = $port.ReadExisting()
Write-Host $response
$port.Close()
Write-Host "Settings updated. Try sending a new test SMS now." -ForegroundColor Green
